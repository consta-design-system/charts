# MixPlot

<RemadeInfo commponentName="MixPlot" apiUrl="https://g2plot.antv.vision/en/docs/api/advanced-plots/mix" />

MixPlot подходит в том случае если вам необходимо сгрупирировать графики по какому-либо 
признаку.

- [Список свойств](#свойства)
- [Примеры](#примеры)

## Свойства

<RemadeTableInfo apiUrl="https://g2plot.antv.vision/en/docs/api/advanced-plots/mix" />

| Свойство | Тип и значения | По умолчанию | Описание                                                       |
| -------- | -------------- | ------------ | -------------------------------------------------------------- |
| `views?` | `IView[]`     | -             | Массив объектов с конфигурацией и данными о геометрии графика  |
| `plots?` | `IPlot[]`     | -             | Массив объектов с даннми о графиках                            |

### Примеры

<MixPlotExample />

```tsx
import { MixOptions } from '@antv/g2plot'

const data = [
  ['Cosmopolitan', 51, 45, 6],
  ['Martini', 67, 39, 28],
  ['Mojito', 19, 11, 8],
  ['Margarita', 47, 33, 14],
  ['Mai Tai', 32, 20, 12],
  ['Beer', 70, 20, 50],
]

const yearData = [
  ['2010', 60, 176, 35, 25],
  ['2011', 51, 136, 25, 26],
  ['2012', 73, 196, 35, 38],
  ['2013', 84, 315, 43, 41],
  ['2014', 79, 203, 36, 33],
  ['2015', 89, 286, 41, 48],
]

const views: MixOptions['views'] = [
  {
    data: data.map(item => ({ type: item[0], value: item[1] })),
    region: { start: { x: 0, y: 0 }, end: { x: 0.5, y: 0.4 } },
    coordinate: {
      type: 'theta',
      cfg: { radius: 0.85 },
    },
    axes: {
      value: { title: { text: 'Drinks' }, grid: null, tickLine: null, line: null, ticks: [] },
    },
    geometries: [
      {
        type: 'interval',
        xField: '1',
        yField: 'value',
        colorField: 'type',
        mapping: {},
        adjust: { type: 'stack' },
      },
    ],
    interactions: [{ type: 'element-active' }, { type: 'association-highlight' }],
  },
  {
    data: [
      ...data.map(item => ({ type: item[0], gender: 'male', value: item[2] })),
      ...data.map(item => ({ type: item[0], gender: 'female', value: item[3] })),
    ],
    region: { start: { x: 0.5, y: 0 }, end: { x: 1, y: 0.45 } },
    axes: { value: false },
    geometries: [
      {
        type: 'interval',
        xField: 'type',
        yField: 'value',
        colorField: 'gender',
        mapping: {},
        adjust: { type: 'dodge', marginRatio: 0 },
      },
    ],
  },
  {
    data: yearData.map(item => ({ year: item[0], ordered: item[1] })),
    region: { start: { x: 0, y: 0.52 }, end: { x: 0.48, y: 1 } },
    axes: {
      year: {
        title: { text: 'Drinks ordered' },
      },
    },
    meta: {
      ordered: {
        min: 40,
        max: 90,
      },
    },
    geometries: [
      {
        type: 'area',
        xField: 'year',
        yField: 'ordered',
        mapping: {},
      },
      {
        type: 'line',
        xField: 'year',
        yField: 'ordered',
        mapping: {
          style: { lineWidth: 0.5 },
        },
      },
    ],
  },
  {
    data: [
      ...yearData.map(item => ({ year: item[0], gender: 'male', turnout: item[3] })),
      ...yearData.map(item => ({ year: item[0], gender: 'female', turnout: item[4] })),
    ],
    region: { start: { x: 0.52, y: 0.52 }, end: { x: 1, y: 1 } },
    axes: {
      year: {
        title: { text: 'Turnout by gender' },
      },
    },
    geometries: [
      {
        type: 'interval',
        xField: 'year',
        yField: 'turnout',
        colorField: 'gender',
        adjust: { type: 'dodge', marginRatio: 0 },
        mapping: {},
      },
    ],
    interactions: [{ type: 'element-active' }, { type: 'association-sibling-highlight' }],
  },
]

export const MixPlotExample = () => {
  return <MixPlot views={views} style={{ height: 500, padding: 'auto' }} />
}
```
