# DualAxes

<RemadeInfo
  commponentName="DualAxes"
  apiUrl="https://g2plot.antv.vision/en/docs/api/plots/dual-axes"
/>

Линейная диаграмма подходит, если вам нужно описать непрерывный процесс (или несколько)
с большим количеством значений. На такой диаграмме хорошо видно, как значения меняются,
и удобно сравнивать изменения в разных процессах.

### Две линии

```tsx
export const data = [
  { year: '1991', value: 3, count: 10 },
  { year: '1992', value: 4, count: 4 },
  { year: '1993', value: 3.5, count: 5 },
  { year: '1994', value: 5, count: 5 },
  { year: '1995', value: 4.9, count: 4.9 },
  { year: '1996', value: 6, count: 35 },
  { year: '1997', value: 7, count: 7 },
  { year: '1998', value: 9, count: 1 },
  { year: '1999', value: 13, count: 20 },
]

export function DualAxesExampleDualLine() {
  return <DualAxes data={[data, data]} xField="year" yField={['value', 'count']} />
}
```

### Две линии со стилизацией

```tsx
export const data = [
  { year: '1991', value: 3, count: 10 },
  { year: '1992', value: 4, count: 4 },
  { year: '1993', value: 3.5, count: 5 },
  { year: '1994', value: 5, count: 5 },
  { year: '1995', value: 4.9, count: 4.9 },
  { year: '1996', value: 6, count: 35 },
  { year: '1997', value: 7, count: 7 },
  { year: '1998', value: 9, count: 1 },
  { year: '1999', value: 13, count: 20 },
]

export function DualAxesExampleDualLineWithStyle() {
  const vars = useThemeVars()
  return (
    <DualAxes
      data={[data, data]}
      xField="year"
      yField={['value', 'count']}
      geometryOptions={[
        {
          geometry: 'line',
          smooth: false,
          color: vars.color.primary['--color-bg-brand'],
          label: {
            formatter: datum => {
              return `${datum.value}`
            },
          },
          lineStyle: {
            lineWidth: 3,
            lineDash: [5, 5],
          },
        },
        {
          geometry: 'line',
          smooth: true,
          color: vars.color.primary['--color-typo-success'],
          lineStyle: {
            lineWidth: 4,
            opacity: 0.5,
          },
          label: {
            formatter: datum => {
              return `${datum.count}`
            },
          },
          point: {
            shape: 'circle',
            size: 4,
            style: {
              opacity: 0.5,
              stroke: vars.color.primary['--color-typo-success'],
              fill: vars.color.primary['--color-bg-default'],
            },
          },
        },
      ]}
    />
  )
}
```

### Линия и колонки

```tsx
const uvBillData = [
  { time: '2019-03', value: 350, type: 'uv' },
  { time: '2019-04', value: 900, type: 'uv' },
  { time: '2019-05', value: 300, type: 'uv' },
  { time: '2019-06', value: 450, type: 'uv' },
  { time: '2019-07', value: 470, type: 'uv' },
  { time: '2019-03', value: 220, type: 'bill' },
  { time: '2019-04', value: 300, type: 'bill' },
  { time: '2019-05', value: 250, type: 'bill' },
  { time: '2019-06', value: 220, type: 'bill' },
  { time: '2019-07', value: 362, type: 'bill' },
]

const transformData = [
  { time: '2019-03', count: 800 },
  { time: '2019-04', count: 600 },
  { time: '2019-05', count: 400 },
  { time: '2019-06', count: 380 },
  { time: '2019-07', count: 220 },
]

export function DualAxesExampleStackedColumnLine() {
  return (
    <DualAxes
      data={[uvBillData, transformData]}
      xField="time"
      yField={['value', 'count']}
      geometryOptions={[
        {
          geometry: 'column',
          isStack: true,
          seriesField: 'type',
        },
        {
          geometry: 'line',
        },
      ]}
    />
  )
}
```
