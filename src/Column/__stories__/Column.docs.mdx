### Пример

```tsx
type Item = {
  name: string
  month: string
  value: number
}

const data: Item[] = [
  {
    name: 'London',
    month: 'Jan.',
    value: 18.9,
  },

  // ...
  {
    name: 'London',
    month: 'Feb.',
    value: 28.8,
  },
]

export function ColumnExample() {
  return <Column data={data} xField="month" yField="value" seriesField="name" />
}
```
